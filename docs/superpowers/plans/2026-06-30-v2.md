```java
package com.icbc.sh.techmg.framework.log;

import com.google.gson.Gson;
import com.icbc.sh.techmg.common.event.OperationLogEvent;
import jakarta.servlet.http.HttpServletRequest;
import lombok.extern.slf4j.Slf4j;
import org.aspectj.lang.ProceedingJoinPoint;
import org.aspectj.lang.annotation.Around;
import org.aspectj.lang.annotation.Aspect;
import org.aspectj.lang.reflect.MethodSignature;
import org.springframework.beans.factory.annotation.Value;
import org.springframework.context.ApplicationEventPublisher;
import org.springframework.stereotype.Component;
import org.springframework.web.context.request.RequestContextHolder;
import org.springframework.web.context.request.ServletRequestAttributes;

import java.util.Arrays;
import java.util.Objects;

@Slf4j
@Aspect
@Component
public class ApiAccessLogAspect {

    private final Gson gson = new Gson();
    private final ApplicationEventPublisher eventPublisher;

    @Value("${api.access-log.max-length:4000}")
    private int maxLogLength;

    public ApiAccessLogAspect(ApplicationEventPublisher eventPublisher) {
        this.eventPublisher = eventPublisher;
    }

    @Around("@annotation(com.icbc.sh.techmg.framework.log.ApiAccessLog)")
    public Object around(ProceedingJoinPoint point) throws Throwable {
        ServletRequestAttributes attributes = (ServletRequestAttributes) RequestContextHolder.getRequestAttributes();
        HttpServletRequest request = attributes != null ? attributes.getRequest() : null;

        // ---- 提取请求元数据 ----
        String httpMethod = request != null ? request.getMethod() : "UNKNOWN";
        String uri = request != null ? request.getRequestURI() : "UNKNOWN";
        String queryString = request != null ? request.getQueryString() : null;
        String fullUrl = uri + (queryString != null ? "?" + queryString : "");
        String ip = getClientIp(request);
        String userAgent = request != null ? request.getHeader("User-Agent") : "UNKNOWN";
        if (userAgent != null && userAgent.length() > 200) {
            userAgent = userAgent.substring(0, 200) + "...";
        }

        MethodSignature signature = (MethodSignature) point.getSignature();
        String className = signature.getDeclaringType().getSimpleName();
        String methodName = signature.getName();

        String operator = "anonymous";
        var auth = org.springframework.security.core.context.SecurityContextHolder.getContext().getAuthentication();
        if (auth != null && !"anonymousUser".equals(auth.getName())) {
            operator = auth.getName();
        }

        // ---- 序列化请求参数 ----
        String requestParam = serializeArgs(point.getArgs());

        // ---- 执行目标方法 ----
        long startTime = System.currentTimeMillis();
        Object result = null;
        Throwable businessException = null;
        int status = 1;
        try {
            result = point.proceed();
        } catch (Throwable e) {
            status = 0;
            businessException = e;
        }
        long costTime = System.currentTimeMillis() - startTime;

        // ---- 序列化响应 ----
        String responseResult;
        if (status == 1) {
            responseResult = serializeResult(result);
        } else {
            responseResult = "error: " + businessException.getMessage();
            if (businessException.getStackTrace().length > 0) {
                StackTraceElement[] stack = businessException.getStackTrace();
                StringBuilder sb = new StringBuilder(responseResult);
                int limit = Math.min(3, stack.length);
                for (int i = 0; i < limit; i++) {
                    sb.append("\n        at ").append(stack[i].toString());
                }
                if (stack.length > 3) {
                    sb.append("\n        ...[stack truncated, total: ").append(stack.length).append(" frames]");
                }
                responseResult = sb.toString();
            }
        }

        // ---- 结构化日志输出 ----
        String logLine = buildLogLine(httpMethod, fullUrl, ip, userAgent, operator,
                requestParam, responseResult, status, costTime);

        if (status == 1) {
            log.info(logLine);
        } else {
            log.error(logLine);
        }

        // ---- 发布操作日志事件 ----
        publishEvent(className, methodName, operator, truncate(requestParam, maxLogLength),
                truncate(responseResult, maxLogLength), ip, costTime, status);

        if (businessException != null) {
            throw businessException;
        }
        return result;
    }

    private String buildLogLine(String httpMethod, String fullUrl, String ip, String userAgent,
                                 String operator, String requestParam, String responseResult,
                                 int status, long costTime) {
        StringBuilder sb = new StringBuilder();
        sb.append("\n[API] ").append(httpMethod).append(" ").append(fullUrl)
                .append(" | IP: ").append(ip)
                .append(" | UA: ").append(userAgent)
                .append(" | Operator: ").append(operator)
                .append("\n     Request: ").append(truncate(requestParam, maxLogLength));

        if (status == 1) {
            int size = responseResult != null ? responseResult.getBytes(java.nio.charset.StandardCharsets.UTF_8).length : 0;
            sb.append("\n     Response(code=200, size=").append(size).append("B): ")
                    .append(truncate(responseResult, maxLogLength));
        } else {
            sb.append("\n     ERROR: ").append(responseResult);
        }
        sb.append("\n     Cost: ").append(costTime).append("ms");
        return sb.toString();
    }

    private String serializeArgs(Object[] args) {
        try {
            if (args == null || args.length == 0) return "{}";
            // 过滤掉 HttpServletRequest/Response 等不可序列化对象
            Object[] filtered = Arrays.stream(args)
                    .filter(Objects::nonNull)
                    .filter(a -> !(a instanceof HttpServletRequest)
                            && !(a instanceof jakarta.servlet.http.HttpServletResponse))
                    .toArray();
            if (filtered.length == 0) return "{}";
            return gson.toJson(filtered.length == 1 ? filtered[0] : filtered);
        } catch (Exception e) {
            return "[serialization error: " + e.getMessage() + "]";
        }
    }

    private String serializeResult(Object result) {
        if (result == null) return "null";
        try {
            return gson.toJson(result);
        } catch (Exception e) {
            return "[serialization error: " + e.getMessage() + "]";
        }
    }

    private String truncate(String text, int maxLen) {
        if (text == null) return "null";
        if (text.length() <= maxLen) return text;
        return text.substring(0, maxLen) + "...[truncated, total: " + text.length() + " chars]";
    }

    private String getClientIp(HttpServletRequest request) {
        if (request == null) return "UNKNOWN";
        String ip = request.getHeader("X-Forwarded-For");
        if (ip == null || ip.isBlank() || "unknown".equalsIgnoreCase(ip)) {
            ip = request.getHeader("X-Real-IP");
        }
        if (ip == null || ip.isBlank() || "unknown".equalsIgnoreCase(ip)) {
            ip = request.getRemoteAddr();
        }
        return ip != null ? ip.split(",")[0].trim() : "UNKNOWN";
    }

    private void publishEvent(String module, String operation, String operator,
                              String requestParam, String responseResult,
                              String ip, long costTime, int status) {
        try {
            eventPublisher.publishEvent(new OperationLogEvent(this,
                    module, operation, operator,
                    requestParam, responseResult, ip, costTime, status));
        } catch (Exception e) {
            log.warn("Failed to publish OperationLogEvent: {}", e.getMessage());
        }
    }
}
```

private final SsoAuthProvider ssoAuthProvider;

// 在构造函数中增加 SsoAuthProvider 参数注入

/**
 * 获取 SSO 登录地址
 */
@GetMapping("/sso/login-url")
public R<Map<String, String>> getSsoLoginUrl() {
    if (ssoAuthProvider == null || !ssoAuthProvider.enabled()) {
        return R.fail(400, "SSO 登录未启用");
    }
    String loginUrl = ssoAuthProvider.getLoginUrl();
    Map<String, String> data = new HashMap<>();
    data.put("loginUrl", loginUrl);
    return R.ok(data);
}

/**
 * SSO ticket 登录
 */
@PostMapping("/sso/login")
@ApiAccessLog
public R<Map<String, Object>> ssoLogin(@RequestBody Map<String, String> body) {
    String ticket = body.get("ticket");
    if (ticket == null || ticket.isBlank()) {
        return R.fail(400, "缺少 SSO ticket");
    }
    if (ssoAuthProvider == null || !ssoAuthProvider.enabled()) {
        return R.fail(400, "SSO 登录未启用");
    }

    // 1. 验证 SSO ticket，获取 userId
    String userId = ssoAuthProvider.authenticate(ticket);
    if (userId == null) {
        return R.fail(401, "SSO 认证失败");
    }

    // 2. 查询用户扩展信息（TODO: 外部 API，目前硬编码）
    Map<String, Object> extInfo;
    if (ssoAuthProvider instanceof SsoAuthProviderImpl impl) {
        extInfo = impl.queryUserExtInfo(userId);
    } else {
        extInfo = Map.of();
    }

    // 3. 更新/插入 sys_user
    SysUser user = new SysUser();
    user.setAuthNo(userId);
    user.setRealName((String) extInfo.getOrDefault("tellername", ""));
    user.setAdAccount((String) extInfo.getOrDefault("ad", ""));
    user.setBranchId((String) extInfo.getOrDefault("branchId", ""));
    user.setBranchName((String) extInfo.getOrDefault("branchName", ""));
    user.setNotesId((String) extInfo.getOrDefault("notesId", ""));
    user.setUsername((String) extInfo.getOrDefault("tellername", ""));
    sysUserService.updateLoginInfo(user);

    // 4. 加载用户角色
    var userDetails = userDetailsService.loadUserByUsername(userId);
    String roles = userDetails.getAuthorities().stream()
            .map(Object::toString)
            .collect(Collectors.joining(","));

    // 5. 生成 JWT

    userInfo.put("branchId", extInfo.getOrDefault("branchId", ""));
    userInfo.put("branchName", extInfo.getOrDefault("branchName", ""));
    userInfo.put("notesId", extInfo.getOrDefault("notesId", ""));
    userInfo.put("branchIdList", extInfo.getOrDefault("branchIdList", java.util.List.of()));

```java
@Override
public int importItems(MultipartFile file, Long subtaskId, String mode, Map<String, Integer> mapping) {
    // 1. 解析文件（Apache POI for xlsx, 或 opencsv for csv）
    List<String> headers = parseHeaders(file);
    List<List<String>> rows = parseRows(file);

    // 2. 自动匹配表头 → 字段名
    Map<String, Integer> columnMap = mapping != null ? mapping : autoMapHeaders(headers);
    // autoMapHeaders: 按列名模糊匹配
    // "应用名"/"应用"/"应用名称" → applicationName
    // "治理项"/"治理条目" → governanceItem
    // "问题描述"/"问题" → issueDescription
    // ...

    // 3. 构建治理条目对象列表
    List<TechReformItem> items = buildItems(rows, columnMap, subtaskId);

    // 4. 根据模式写入
    if ("OVERWRITE".equalsIgnoreCase(mode)) {
        // 全量覆盖：删除旧数据，全部插入
        remove(new LambdaQueryWrapper<TechReformItem>()
                .eq(TechReformItem::getSubtaskId, subtaskId));
        saveBatch(items);
    } else {
        // Merge 模式：按 applicationName + governanceItem 唯一键比对
        for (TechReformItem item : items) {
            TechReformItem existing = getOne(new LambdaQueryWrapper<TechReformItem>()
                    .eq(TechReformItem::getSubtaskId, subtaskId)
                    .eq(TechReformItem::getApplicationName, item.getApplicationName())
                    .eq(TechReformItem::getGovernanceItem, item.getGovernanceItem()));
            if (existing == null) {
                save(item);
            } else {
                // 保留已反馈内容
                if (existing.getFeedback() != null && !existing.getFeedback().isBlank()) {
                    item.setFeedback(existing.getFeedback());
                }
                if (existing.getGovernancePlan() != null && !existing.getGovernancePlan().isBlank()) {
                    item.setGovernancePlan(existing.getGovernancePlan());
                }
                item.setId(existing.getId());
                updateById(item);
            }
        }
    }
    return items.size();
}
```
